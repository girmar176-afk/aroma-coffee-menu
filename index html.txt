<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Coffee Shop QR Scanner</title>
  <script src="https://unpkg.com/html5-qrcode" type="text/javascript"></script>
  <link rel="stylesheet" href="style.css">
</head>
<body>
  <h1>☕ Welcome to Aroma Coffee</h1>
  <p>Scan the QR on your table to view the menu</p>

  <div id="reader"></div>
  <div id="result"></div>

  <script>
    function onScanSuccess(decodedText) {
      // If the QR contains "menu", go to that page
      if (decodedText.includes("menu.html")) {
        window.location.href = decodedText;
      } else {
        document.getElementById('result').innerHTML =
          `✅ Scanned: <a href="${decodedText}" target="_blank">${decodedText}</a>`;
      }
    }

    function onScanError(errorMessage) {
      // Ignore errors
    }

    let html5QrcodeScanner = new Html5QrcodeScanner(
      "reader",
      { fps: 10, qrbox: 250 }
    );
    html5QrcodeScanner.render(onScanSuccess, onScanError);
  </script>
</body>
</html>
